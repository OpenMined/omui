## Colors

<Theme>
  {Object.keys(colors).map((color) => (
    <Flex justifyContent="space-between" alignItems="center" mt={3} key={color}>
      <Text width="80px" fontWeight="bold">
        {color[0].toUpperCase() + color.slice(1)}
      </Text>
      {Object.keys(colors[color]).map((num) => (
        <Flex
          key={`${color}-${num}`}
          width="80px"
          height="80px"
          bg={`${color}.${num}`}
          borderRadius="lg"
          justifyContent="center"
          alignItems="center"
        >
          <Text
            color={
              isAccessible('gray.900', `${color}.${num}`)(theme)
                ? 'gray.900'
                : 'white'
            }
          >
            {num}
          </Text>
        </Flex>
      ))}

  ))}
